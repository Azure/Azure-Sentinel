// Parser: NetskopeAlertEvents
// Normalizes Netskope Alerts & Events ingested into NetskopeAlertEvents_CL.
// Column names match the published Netskope Log Streaming connector schema.
// Deployed as a saved function (alias: NetskopeAlertEvents) by the solution package.
let NetskopeAlertEvents = () {
    NetskopeAlertEvents_CL
    | extend
        alert_id = tostring(column_ifexists('AlertId', '')),
        event_type = tostring(column_ifexists('NsType', '')),
        user = tostring(column_ifexists('User', '')),
        user_ip = tostring(column_ifexists('Userip', '')),
        application = tostring(column_ifexists('App', '')),
        severity = tostring(column_ifexists('Severity', '')),
        policy = tostring(column_ifexists('Policy', '')),
        action = tostring(column_ifexists('Action', '')),
        device = tostring(column_ifexists('Device', '')),
        location = iff(isempty(tostring(column_ifexists('Location', ''))),
                       strcat_array(pack_array(tostring(column_ifexists('SrcLocation', '')), tostring(column_ifexists('SrcCountry', ''))), ', '),
                       tostring(column_ifexists('Location', ''))),
        raw_json = tostring(column_ifexists('RawData', ''))
    | project
        TimeGenerated,
        alert_id,
        event_type,
        user,
        user_ip,
        application,
        severity,
        policy,
        action,
        device,
        location,
        raw_json
};
